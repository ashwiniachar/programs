#include<stdio.h>
int main()
{  
   int N, i,c[100],temp,sum;
   printf("Enter the number of coders\n");
   scanf("%d",&N);
   printf("Enter the skill rates of each coders\n");
   for(i=0;i<N;i++)
   {
      scanf("%d",&c[i]);
   }
   for(i=0;i<N;i++)
   {  
      if(c[i]>c[i+1])
       { 
         temp=c[i];
         c[i]=c[i+1];
         c[i]=temp;
       }
   }
   for(i=N;i>N/2;i--)
   {
      sum=sum+c[i];
   }
   printf("%d",sum);
}
